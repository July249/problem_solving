```js
function solution(num_list) {
  var even = 0;
  var odd = 0;

  num_list.forEach((num) => (num % 2 === 0 ? (even += 1) : (odd += 1)));

  var answer = [even, odd];
  return answer;
}
```

```py
def solution(num_list) {
  even = 0
  odd = 0
  for i in num_list:

      even += 1
    else:
      odd += 1
  return [even, odd]
}
```
